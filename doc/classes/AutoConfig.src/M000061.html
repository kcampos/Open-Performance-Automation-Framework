  def parse_suite(suite=self.suite)
    
    self.tests = {}
    
    File.open("#{self.suite_dir}/#{suite}", "r") do |suite_file|
      while(line = suite_file.gets)
        next if(line =~ /^\#/) # skip if it's a comment
        line =~ /^(\w+\.rb)\,(\d+)$/
        self.tests[$1] = $2
      end
    end
    
  end
